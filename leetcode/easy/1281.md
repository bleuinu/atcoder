# 1281. Subtract the Product and Sum of Digits of an Integer

```cpp
class Solution {
public:
    int subtractProductAndSum(int n) {
        int prod = n%10;
        int sum = n%10;
        n/=10;
        
        while(n>0) {
            int t = n%10;
            prod *= t;
            sum += t;
            n/=10;
        }
        
        return prod - sum;
    }
};
```
